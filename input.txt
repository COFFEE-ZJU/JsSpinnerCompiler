deptCnt = streamT -> window [range now] -> group each ele by d = ele.dept into {department:d, empCnt:count($), avgSalary: avg($[*].salary)};
emps = streamT->window [range 3 hours] -> filter $.salary*12>=1000000 or $.is_manager;

view3 = join deptCnt,emps where emps.dept == deptCnt.department into {emps.name, emps.salary, avgInDept:deptCnt.avgSalary, emps.another_object.tags};
